                                                  function formatDate(timestamp, fmt="yyyy/MM/dd hh:mm:ss") {
    const date = new Date(timestamp * 1000); // å°†æ—¶é—´æˆ³ä¹˜ä»¥1000è½¬æ¢ä¸ºæ¯«ç§’ï¼Œå¹¶åˆ›å»º Date å¯¹è±¡

    const options = {
        year: 'numeric',
        month: '2-digit',
        day: '2-digit',
        hour: '2-digit',
        minute: '2-digit',
        second: '2-digit'
    };

    fmt = fmt.replace('yyyy', date.toLocaleString('en', { year: 'numeric' })) // æ›¿æ¢å¹´ä»½
        .replace('MM', date.toLocaleString('en', { month: '2-digit' })) // æ›¿æ¢æœˆä»½
        .replace('dd', date.toLocaleString('en', { day: '2-digit' })) // æ›¿æ¢æ—¥æœŸ
        .replace('hh', date.toLocaleString('en', { hour: '2-digit', hour12: false })) // æ›¿æ¢å°æ—¶
        .replace('mm', date.toLocaleString('en', { minute: '2-digit' })) // æ›¿æ¢åˆ†é’Ÿ
        .replace('ss', date.toLocaleString('en', { second: '2-digit' })); // æ›¿æ¢ç§’æ•°

    return fmt;
}
